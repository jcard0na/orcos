57void WakeUpTimerEventCallback(RTC_HandleTypeDef *hrtc)
58{
59 // DEBUG_PRINT("-- LPTIM1: EXTIN toggle ---\n");
60 RTC_TimeTypeDef Time;
61 RTC_DateTypeDef Date;
62 /* Get the RTC calendar time */
63 HAL_RTC_GetTime(hrtc, &Time, RTC_FORMAT_BIN);
64 // Do not skip reading the date: This triggers an update to the shadow registers!
65 HAL_RTC_GetDate(hrtc, &Date, RTC_FORMAT_BIN);
66#if DEBUG
67 SEGGER_RTT_printf(0, "wake up timer event %02d:%02d! (%d)\n", Time.Minutes, Time.Seconds, timeout_counter);
68#endif
69
70 // If we are showng the RTC test screen, update the time
71 if (current_test_screen == 5)
72 {
73 LCD_test_screen(current_test_screen);
74 }
75
76 timeout_counter++;
77 if (timeout_counter > OFF_TIMEOUT)
78 {
79 LCD_power_off(1);
80 }
81}

88void LCD_power_on()
89{
90 DEBUG_PRINT("\n--- LDC_power_on() ---\n");
91 HAL_GPIO_WritePin(GPIOA, GPIO_PIN_15, GPIO_PIN_SET); // 5V booster enable
92 HAL_Delay(1);
93 HAL_GPIO_WritePin(GPIOA, GPIO_PIN_10, GPIO_PIN_SET); // DISP signal to "ON"
94 HAL_TIM_Base_Start_IT(&htim1);
95 /* Configure wakeup interrupt */
96 /* RTC Wakeup Interrupt Generation:
97 (2047 + 1) × (16 / 32768) = 1.000 seconds
98 */
99 if (HAL_RTC_RegisterCallback(&hrtc, HAL_RTC_WAKEUPTIMER_EVENT_CB_ID, WakeUpTimerEventCallback) != HAL_OK)
100 {
101 LCD_Error_Handler();
102 }
103 HAL_RTCEx_SetWakeUpTimer_IT(&hrtc, 2047, RTC_WAKEUPCLOCK_RTCCLK_DIV16, 0);
104 lcd_is_on = true;
105}

107void LCD_power_off(int clear)
108{
109 DEBUG_PRINT("\n--- LDC_power_off() ---\n");
110 // XXX: this prevents waking up form STOP2
111 // HAL_TIM_Base_Stop_IT(&htim1); // Stop the timer
112 delay_us(30);
113 if (clear)
114 HAL_GPIO_WritePin(GPIOA, GPIO_PIN_10, GPIO_PIN_RESET); // DISP signal to "OFF"
115 delay_us(30);
116 HAL_GPIO_WritePin(GPIOA, GPIO_PIN_9, GPIO_PIN_RESET); // EXTCOMIN signal of "OFF"
117 HAL_GPIO_WritePin(GPIOA, GPIO_PIN_15, GPIO_PIN_RESET); // 5V booster disable
118 HAL_RTCEx_DeactivateWakeUpTimer(&hrtc);
119 lcd_is_on = false;
120}

134// Sends line data to LCD.
135//
136// Parameters
137// buf Buffer with line data
138//
139// Buffer should be of LCD_LINE_BUF_SIZE.
140//
141// With following values:
142//
143// [0] - reserved space for LCD write command
144// [1] - LCD line number
145// [2..51] - 50 bytes of LCD line data
146// [52..53] - padding required by LCD hw
147//
148// Only line number and line data have to by filled by user.

151int lcd_for_calc(int what_screen)
152{
153 switch (what_screen)
154 {
155 case DISP_SYS_MENU:
156 lcd_clear_buffer();
157
158 // Get and display RTC time
159 tm_t time;
160 dt_t date;
161 rtc_read(&time, &date);
162
163 char time_str[20];
164 snprintf(time_str, sizeof(time_str), "%02d:%02d:%02d",
165 time.hour, time.min, time.sec);
166 lcd_putsAt(time_str, FONT_16x26, 120, 100, LCD_SET_VALUE);
167
168 char date_str[20];
169 snprintf(date_str, sizeof(date_str), "%02d/%02d/20%02d",
170 date.day, date.month, date.year);
171 lcd_putsAt(date_str, FONT_16x26, 120, 130, LCD_SET_VALUE);
172
173 char voltage_str[20];
174 uint16_t vbat = (uint16_t)get_vbat();
175 snprintf(voltage_str, sizeof(voltage_str), "bat: %01d.%03dV",
176 vbat / 1000, vbat % 1000);
177 lcd_putsAt(voltage_str, FONT_16x26, 0, 0, LCD_SET_VALUE);
178
179 break;
180 default:
181 lcd_draw_test_pattern(8);
182 }
183 return LCD_HEIGHT;
184}

211 lcd_draw_img(rook_img, 32, 32, i * 64, j * 64, LCD_EMPTY_VALUE);

214 lcd_draw_img(rook_img, 32, 32, i * 64 + 32, j * 64, LCD_SET_VALUE);
215 }
216 }
217 }
218 if (count == 4)
219 {
220 lcd_clear_buffer();
221 lcd_draw_img(pixel_data_bin, 400, 240, 0, 0, LCD_SET_VALUE);
222 lcd_draw_img(test_img, 32, 32, 0, 0, LCD_SET_VALUE);
223 lcd_draw_img(test_img, 32, 32, 50, 50, LCD_SET_VALUE);
224 lcd_draw_img(test_img, 32, 32, 90, 90, LCD_EMPTY_VALUE);
225
226 lcd_draw_img((uint8_t[]){0xff, 0xff}, 2, 2, 100, 100, LCD_SET_VALUE);
227 lcd_draw_img((uint8_t[]){0xff, 0xff}, 2, 2, 106, 100, LCD_SET_VALUE);
228 lcd_draw_img((uint8_t[]){0xff, 0xff}, 2, 2, 100, 106, LCD_SET_VALUE);
229 lcd_draw_img((uint8_t[]){0xff, 0xff}, 2, 2, 106, 106, LCD_SET_VALUE);
230 }
231 if (count == 5)
232 {
233 lcd_for_calc(DISP_SYS_MENU);
234 }
235 if (count == 6)
236 {
237 lcd_clear_buffer();
238
239 lcd_putsAt("OpenRPNCalc", FONT_24x40, 72, 40, LCD_SET_VALUE);
240 lcd_putsAt("Open Hardware", FONT_16x26, 96, 80, LCD_SET_VALUE);
241 lcd_putsAt("Hello World!", FONT_12x20, 110, 120, LCD_SET_VALUE);
242 lcd_putsAt("Small and Bold!", FONT_7x12b, 130, 140, LCD_SET_VALUE);
243 lcd_putsAt("tiny, tiny, behold!", FONT_6x8, 125, 160, LCD_SET_VALUE);
244 lcd_putsAt("tiny, tiny, behold!", FONT_6x8, 125, 170, LCD_SET_VALUE);
245 lcd_putsAt("tiny, tiny, behold!", FONT_6x8, 125, 180, LCD_SET_VALUE);
246 lcd_putsAt("tiny, tiny, behold!", FONT_6x8, 125, 190, LCD_SET_VALUE);
247 }
248 if (count == 7)
249 {
250 lcd_clear_buffer();
251 // First line - normal black text
252 lcd_putsAt("Reverse", FONT_24x40, 120, 80, LCD_SET_VALUE);
253
254 // Second line - highlighted text with background
255 FontDef_t *font = font_lookup(FONT_24x40);
256 uint16_t text_width = strlen("Polish") * font->FontWidth;
257 uint16_t text_height = font->FontHeight;
258 uint16_t padding = 10;
259
260 // Draw text first
261 lcd_putsAt("Polish", FONT_24x40, 120, 120, LCD_SET_VALUE);
262
263 // XOR the entire text area (background and text)
264 for (uint16_t y = 120 - padding / 2; y < 120 + text_height + padding / 2; y++)
265 {
266 uint16_t remaining_width = text_width + padding;
267 uint16_t x_pos = 120 - padding / 2;
268 while (remaining_width > 0)
269 {
270 uint16_t chunk_width = (remaining_width > 24) ? 24 : remaining_width;
271 bitblt24(x_pos, chunk_width, y, 0xFFFFFFFF, BLT_XOR, BLT_NONE);
272 x_pos += chunk_width;
273 remaining_width -= chunk_width;
274 }
275 }
276
277 // Add rectangle demonstration
278 int color = LCD_SET_VALUE;
279 for (int i = 0; i < 4; i++)
280 {
281 lcd_fill_rect(10 + 10 * i, 10 + 5 * i, 70 - 20 * i, 70 - 20 * i, color);
282 color = ~color;
283 }
284 }
285 if (count == 8)
286 {
287 // Draw test image line by line using LCD_write_line
288 uint8_t line_buffer[LCD_LINE_BUF_SIZE] = {0};
289
290 // Loop through each line of the test image
291 for (int y = 0; y < 240; y++)
292 {
293 // Set line number (1-based)
294 line_buffer[1] = y + 1;
295
296 // Copy one line swapping byte order
297 for (int x = 0; x < LCD_LINE_SIZE; x++)
298 {
299 uint8_t original = pixel_data_bin[y * LCD_LINE_SIZE + x];
300 // Reverse bits in each byte
301 line_buffer[2 + x] = ((original & 0x01) << 7) |
302 ((original & 0x02) << 5) |
303 ((original & 0x04) << 3) |
304 ((original & 0x08) << 1) |
305 ((original & 0x10) >> 1) |
306 ((original & 0x20) >> 3) |
307 ((original & 0x40) >> 5) |
308 ((original & 0x80) >> 7);
309 line_buffer[2 + x] = ~line_buffer[2 + x];
310 }
311
312 // Send the line
313 LCD_write_line(line_buffer);
314 }
315 }
316 // Screens that call LCD_write_line(line_buffer) do not need refresh
317 if (count != 8)
318 lcd_refresh();
319}
