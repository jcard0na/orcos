110void lcd_refresh()
111{
112#define CHUNK_SIZE_IN_LINES 240
113#define CHUNK_BUFFER_SIZE (1 + CHUNK_SIZE_IN_LINES * (1 + LCD_WIDTH / 8 + 1) + 1)
114 static uint8_t frame_buffer[CHUNK_BUFFER_SIZE];
115
116 const int total_lines = LCD_HEIGHT;
117 const int chunk_size = CHUNK_SIZE_IN_LINES;
118 const int num_chunks = (total_lines + chunk_size - 1) / chunk_size;
119
120 uint8_t nop = 0x00;
121 HAL_GPIO_WritePin(GPIOC, GPIO_PIN_0, GPIO_PIN_SET);
122 delay_us(10);
123 HAL_SPI_Transmit(&hspi2, &nop, 1, HAL_MAX_DELAY);
124 delay_us(10);
125 HAL_GPIO_WritePin(GPIOC, GPIO_PIN_0, GPIO_PIN_RESET);
126 delay_us(10);
127
128 for (int chunk = 0; chunk < num_chunks; chunk++)
129 {
130 int pos = 0;
131 int start_line = chunk * chunk_size;
132 int lines_in_chunk = (chunk == num_chunks - 1) ? (total_lines - start_line) : chunk_size;
133
134 frame_buffer[pos++] = 0x01;
135 for (int y = 0; y < lines_in_chunk; y++)
136 {
137 frame_buffer[pos++] = start_line + y + 1;
138 memcpy(&frame_buffer[pos], g_framebuffer[start_line + y], LCD_WIDTH / 8);
139 pos += LCD_WIDTH / 8;
140 frame_buffer[pos++] = 0x00;
141 }
142 frame_buffer[pos++] = 0x00;
143
144 HAL_GPIO_WritePin(GPIOC, GPIO_PIN_0, GPIO_PIN_SET);
145 delay_us(12);
146 HAL_SPI_Transmit(&hspi2, frame_buffer, pos, HAL_MAX_DELAY);
147 delay_us(4);
148 HAL_GPIO_WritePin(GPIOC, GPIO_PIN_0, GPIO_PIN_RESET);
149 delay_us(4);
150 }
151}
