535double convert_input()
536{
537 int i;
538 double number = 0.;
539 double shift = 1;
540
541 for (i = 0; i < input.mpos; i++)
542 {
543 number += input.mantissa[input.mpos - i - 1] * shift;
544 shift *= 10;
545 }
546 int exponent = 100 * input.exponent[2] + 10 * input.exponent[1] + input.exponent[0];
547 if (input.expsign)
548 exponent = -exponent;
549 if (input.point)
550 exponent -= (input.mpos - input.point);
551
552 number *= pow(10, exponent);
553 if (input.sign)
554 number = -number;
555
556 if (!finite(number))
557 {
558 draw_flags |= DRAW_ERROR;
559 error_flag = 1;
560 }
561 return (number);
562}

1098 break;

1101 break;
1102 case OP_PYX:
1103 f = exp(lgamma_r(y + 1, &gamma_sign) - lgamma_r(y - x + 1, &gamma_sign));
1104 break;
1105 case OP_SIGNIF_XY:
1106 if (context != CONTEXT_UNCERT)
1107 return;

1111 case OP_CHI2_PROB:
1112 f = chisquared_cdf_c(x, y);
1113 break;
1114 default:
1115 break;
1116 }

1118 if (context == CONTEXT_UNCERT)
1119 {
1120 double ex = stack2[0];
1121 double ey = stack2[1];
1122 switch (code)
1123 {
1124 case OP_PLUS:
1125 case OP_MINUS:
1126 ef = sqrt(ex * ex + ey * ey);
1127 break;
1128 case OP_MULT:
1129 ef = sqrt(x * x * ey * ey + y * y * ex * ex);
1130 break;
1131 case OP_DIV:
1132 ef = sqrt(ey * ey / x / x + ex * ex / x / x / x / x * y * y);
1133 break;
1134 case OP_POW:
1135 ef = sqrt(ex * ex * pow(log(y) * pow(y, x), 2) + ey * ey * pow(x * pow(y, x - 1), 2));
1136 break;
1137 case OP_ROOTX:

1142 break;

1144 break;

1425 else

1435 // set_clock_1mhz();
1436 maybe_convert_input();

1438 for (uint16_t step = 0; step < program_size; step++)
1439 {
1440 uint16_t c = program[step];
1441 uint16_t opmode = c & OPMODE_MASK;
1442 if (opmode != OPMODE_PROG) // Make sure we don't enter infinite nested loop
1443 apply_op(c);
1444 }
1445 draw_flags &= ~(DRAW_INPUT | DRAW_CURSOR);
1446 // set_clock_8mhz();
1447 break;
1448 case OP_STEP:
1449 if (program_entry)
1450 return;
1451 if (program_step == 0)
1452 { // Convert input only at the start of the program
1453 maybe_convert_input();
1454 input.replace_x = 0;
1455 }
1456 if (program_step < program_size)
1457 {
1458 uint16_t c = program[program_step];
1459 uint16_t opmode = c & OPMODE_MASK;
1460 if (opmode != OPMODE_PROG) // Make sure we don't enter infinite nested loop
1461 apply_op(c);
1462 program_step++;
1463 }
1464 if (program_step >= program_size)
1465 program_step = 0; // Rewind to start
1466 break;
1467 case OP_REWIND:
1468 if (program_entry)
1469 return;
1470 program_step = 0;
1471 break;
1472
1473 default:
1474 break;

1478// Process the operation given its code
1479void apply_op(uint16_t code)
1480{
1481 uint16_t opmode = code & OPMODE_MASK;
1482 if (program_entry && opmode != OPMODE_PROG)
1483 {
1484 // Recording program
1485 if (program_size < MAX_PROGRAM_SIZE)
1486 {
1487 program[program_size] = code;
1488 program_size++;
1489 }
1490 else
1491 {
1492 // Exceeded maximum program size
1493 program_entry = 0;
1494 }
1495 draw_flags |= DRAW_STATUS;
1496 }
1497
1498 if (code == OP_NOP)
1499 return;
1500 if (opmode == OPMODE_ENTER)
1501 {
1502 apply_enter(code);
1503 return;
1504 }
1505 if (opmode == OPMODE_STACK)
1506 {
1507 apply_stack(code);
1508 draw_flags |= DRAW_STACK;
1509 input_uncert = 0;
1510 return;
1511 }
1512 if (opmode == OPMODE_CONST)
1513 {
1514 apply_const(code);
1515 draw_flags |= DRAW_STACK;
1516 input_uncert = 0;
1517 return;
1518 }
1519 if (opmode == OPMODE_RCL)
1520 {
1521 apply_memory_rcl();
1522 draw_flags |= DRAW_STACK;
1523 input_uncert = 0;
1524 return;
1525 }
1526 if (opmode == OPMODE_STO)
1527 {
1528 apply_memory_sto();
1529 draw_flags |= DRAW_STATUS;
1530 input_uncert = 0;
1531 return;
1532 }
1533 if (opmode == OPMODE_RCLX)
1534 {
1535 apply_memory_rcl_x();
1536 draw_flags |= DRAW_STACK;
1537 input_uncert = 0;
1538 return;
1539 }
1540 if (opmode == OPMODE_STOX)
1541 {
1542 apply_memory_sto_x();
1543 input_uncert = 0;
1544 return;
1545 }
1546 if (opmode == OPMODE_MPLUS)
1547 {
1548 apply_memory_plus();
1549 draw_flags |= DRAW_STATUS;
1550 input_uncert = 0;
1551 return;
1552 }
1553 if (opmode == OPMODE_MMINUS)
1554 {
1555 apply_memory_minus();
1556 draw_flags |= DRAW_STATUS;
1557 input_uncert = 0;
1558 return;
1559 }
1560 if (opmode == OPMODE_PROG)
1561 {
1562 apply_prog(code);
1563 return;
1564 }
1565
1566 if (input.started)
1567 {
1568 if (context != CONTEXT_UNCERT || input_uncert == 0)
1569 {
1570 stack[0] = convert_input();
1571 }
1572 else
1573 {
1574 stack2[0] = convert_input();
1575 }
1576 input_uncert = 0;
1577 clear_input();
1578 }
1579 if (error_flag)
1580 return;
1581 input.replace_x = 0;
1582 if (opmode == OPMODE_1TO1)
1583 apply_func_1to1(code);
1584 if (opmode == OPMODE_2TO1)
1585 apply_func_2to1(code);
1586 if (opmode == OPMODE_2TO2)
1587 apply_func_2to2(code);
1588 if (opmode == OPMODE_3TO1)
1589 apply_func_3to1(code);
1590 if (opmode == OPMODE_STAT)
1591 apply_func_stat(code);
1592 if (opmode == OPMODE_TEST)
1593 apply_test(code);
1594 else
1595 draw_flags |= DRAW_STACK;
1596}
