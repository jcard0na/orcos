38void lcd_putsAt(const char *str, uint8_t font_id, uint16_t dx, uint16_t dy, uint8_t color)
39{
40 FontDef_t *font = font_lookup(font_id);
41 if (font == NULL)
42 return;
43
44 uint16_t width = font->FontWidth;
45 uint16_t height = font->FontHeight;
46 uint16_t bytes_per_char = (width + 7) / 8;
47 const char (*font_data)[bytes_per_char * height] = font->data;
48
49 if (dy >= LCD_HEIGHT)
50 return;
51
52 uint16_t xpos = ((dx + 7) / 8) << 3;
53 int char_idx = 0;
54
55 // Temporary buffer for one character
56 uint8_t char_buffer[bytes_per_char * height];
57
58 while (xpos < LCD_WIDTH && str[char_idx] != '\0')
59 {
60 uint8_t current_char = str[char_idx];
61 const char *char_data = font_data[current_char];
62
63 // Copy character data directly (already in MSB format)
64 memcpy(char_buffer, char_data, bytes_per_char * height);
65
66 // Draw character using MSB-optimized function
67 lcd_draw_img_unaligned(char_buffer, width, height, xpos, dy, color, true);
68
69 xpos += width;
70 char_idx++;
71 }
72}

89static void lcd_draw_img_aligned(const uint8_t *img, uint32_t w, uint32_t h, uint32_t x, uint32_t y, uint8_t color, bool msb)
90{
91 uint32_t img_stride = w / 8;
92 uint32_t dest_x = x / 8;
93
94 for (uint32_t dy = 0; dy < h; dy++)
95 {
96 uint32_t dest_y = y + dy;
97 if (dest_y >= LCD_HEIGHT)
98 break;
99
100 for (uint32_t sx = 0; sx < img_stride; sx++)
101 {
102 uint8_t src_byte = msb ? img[dy * img_stride + sx] : reverse_bits(img[dy * img_stride + sx]);
103 if (src_byte)
104 {
105 uint32_t dx = dest_x + sx;
106 if (dx < (LCD_WIDTH / 8))
107 {
108 if (color == LCD_SET_VALUE)
109 {
110 g_framebuffer[dest_y][dx] &= ~src_byte; // Clear bits for white
111 }
112 else
113 {
114 g_framebuffer[dest_y][dx] |= src_byte; // Set bits for black
115 }
116 }
117 }
118 }
119 }

122static void lcd_draw_img_unaligned(const uint8_t *img, uint32_t w, uint32_t h, uint32_t x, uint32_t y, uint8_t color, bool msb)
123{
124 uint32_t img_stride = (w + 7) / 8;
125 uint8_t x_align = x % 8;
126
127 for (uint32_t dy = 0; dy < h; dy++)
128 {
129 uint32_t dest_y = y + dy;
130 if (dest_y >= LCD_HEIGHT)
131 continue;
132
133 for (uint32_t sx = 0; sx < img_stride; sx++)
134 {
135 uint8_t img_byte = msb ? img[dy * img_stride + sx] : reverse_bits(img[dy * img_stride + sx]);
136 if (img_byte == 0)
137 continue;
138
139 uint32_t dest_byte = (x / 8) + sx;
140 uint8_t shift = x_align;
141
142 if (dest_byte < (LCD_WIDTH / 8))
143 {
144 uint8_t mask = img_byte << shift;
145 if (color == LCD_SET_VALUE)
146 {
147 g_framebuffer[dest_y][dest_byte] &= ~mask; // Clear bits
148 }
149 else
150 {
151 g_framebuffer[dest_y][dest_byte] |= mask; // Set bits
152 }
153 }
154
155 if (shift != 0 && dest_byte + 1 < (LCD_WIDTH / 8))
156 {
157 uint8_t mask = img_byte >> (8 - shift);
158 if (color == LCD_SET_VALUE)
159 {
160 g_framebuffer[dest_y][dest_byte + 1] &= ~mask;
161 }
162 else
163 {
164 g_framebuffer[dest_y][dest_byte + 1] |= mask;
165 }
166 }
167 }
168 }

171void bitblt24(uint32_t x, uint32_t dx, uint32_t y, uint32_t val, int blt_op, int fill)
172{
173 if (x >= LCD_WIDTH || y >= LCD_HEIGHT || dx == 0 || dx > 24)
174 return;
175
176 // Clamp dx to remaining width
177 if (x + dx > LCD_WIDTH)
178 dx = LCD_WIDTH - x;
179
180 // Apply fill mode to source value
181 if (fill == BLT_SET)
182 {
183 if (blt_op == BLT_OR)
184 {
185 val = 0;
186 }
187 else if (blt_op == BLT_ANDN)
188 {
189 val = ~0;
190 }
191 }
192
193 // Process each bit
194 for (uint32_t i = 0; i < dx; i++)
195 {
196 uint32_t bit_pos = x + i;
197 uint8_t bit_val = (val >> (dx - 1 - i)) & 1;
198 uint8_t *byte_ptr = &g_framebuffer[y][bit_pos / 8];
199 uint8_t bit_mask = 1 << bit_pos % 8;
200
201 switch (blt_op)
202 {
203 case BLT_OR:
204 if (bit_val)
205 {
206 *byte_ptr |= bit_mask; // Set bit (black pixel)
207 }
208 else if (fill == BLT_SET)
209 {
210 *byte_ptr &= ~bit_mask; // Clear bit (white pixel)
211 }
212 break;
213 case BLT_ANDN:
214 if (!bit_val)
215 {
216 *byte_ptr &= ~bit_mask;
217 }
218 else if (fill == BLT_SET)
219 {
220 *byte_ptr |= bit_mask;
221 }
222 break;
223 case BLT_XOR:
224 if (bit_val)
225 {
226 *byte_ptr ^= bit_mask;
227 }
228 break;
229 }
230 }
231}

233uint8_t reverse_bits(uint8_t b)
234{
235 b = (b & 0xF0) >> 4 | (b & 0x0F) << 4;
236 b = (b & 0xCC) >> 2 | (b & 0x33) << 2;
237 b = (b & 0xAA) >> 1 | (b & 0x55) << 1;
238 return b;
239}

241void lcd_fill_rect(uint32_t x, uint32_t y, uint32_t dx, uint32_t dy, int val)
242{
243 // Clamp to screen bounds
244 if (x >= LCD_WIDTH || y >= LCD_HEIGHT)
245 return;
246 if (x + dx > LCD_WIDTH)
247 dx = LCD_WIDTH - x;
248 if (y + dy > LCD_HEIGHT)
249 dy = LCD_HEIGHT - y;
250
251 for (uint32_t curr_y = y; curr_y < y + dy; curr_y++)
252 {
253 for (uint32_t curr_x = x; curr_x < x + dx; curr_x++)
254 {
255 uint8_t *byte_ptr = &g_framebuffer[curr_y][curr_x / 8];
256 uint8_t bit_mask = 1 << (curr_x % 8);
257
258 if (val == LCD_SET_VALUE)
259 {
260 *byte_ptr &= ~bit_mask; // Clear bit (black pixel)
261 }
262 else
263 {
264 *byte_ptr |= bit_mask; // Set bit (white pixel)

281void lcd_draw_test_pattern(uint8_t square_size)
282{
283 // Ensure square_size is at least 1 and not too large
284 if (square_size < 1)
285 square_size = 1;
286 if (square_size > 32)
287 square_size = 32;
288
289 for (int y = 0; y < LCD_HEIGHT; y++)
290 {
291 for (int x = 0; x < LCD_WIDTH; x++)
292 {
293 // Calculate checkerboard pattern
294 uint8_t x_block = x / square_size;
295 uint8_t y_block = y / square_size;
296 uint8_t pattern = (x_block + y_block) % 2;
297
298 // Set pixel in framebuffer
299 if (pattern)
300 {
301 g_framebuffer[y][x / 8] |= (1 << (7 - (x % 8))); // Set pixel white
302 }
303 else
304 {
305 g_framebuffer[y][x / 8] &= ~(1 << (7 - (x % 8))); // Set pixel black
306 }
307 }
308 }
